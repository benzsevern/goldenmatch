**Tech Stack:** Python, aiohttp (optional dep for A2A), sqlite3 (stdlib), existing goldenmatch modules

# goldenmatch/core/review_queue.py
"""Confidence-gated review queue with pluggable storage backends."""
from __future__ import annotations

from dataclasses import dataclass, field
from datetime import datetime


@dataclass
class ReviewItem:
    """A pair held for review."""
    job_name: str
    id_a: int
    id_b: int
    score: float
    explanation: str
    status: str = "pending"
    decided_by: str | None = None
    decided_at: datetime | None = None
    reason: str | None = None

    def approve(self, decided_by: str) -> None:
        self.status = "approved"
        self.decided_by = decided_by
        self.decided_at = datetime.now()

    def reject(self, decided_by: str, reason: str = "") -> None:
        self.status = "rejected"
        self.decided_by = decided_by
        self.decided_at = datetime.now()
        self.reason = reason


def gate_pairs(
    pairs: list[tuple[int, int, float]],
    merge_threshold: float = 0.95,
    review_threshold: float = 0.75,
) -> tuple[list, list, list]:
    """Split scored pairs into auto-merge, review, and auto-reject."""
    auto_merged = []
    review = []
    auto_rejected = []
    for a, b, score in pairs:
        if score > merge_threshold:
            auto_merged.append((a, b, score))
        elif score >= review_threshold:
            review.append((a, b, score))
        else:
            auto_rejected.append((a, b, score))
    return auto_merged, review, auto_rejected


class ReviewQueue:
    """Review queue with pluggable storage."""

    def __init__(self, backend: str = "memory"):
        self._backend = backend
        if backend == "memory":
            self._items: dict[str, list[ReviewItem]] = {}
        elif backend == "sqlite":
            self._init_sqlite()
        elif backend == "postgres":
            pass  # Postgres backend uses DATABASE_URL
        else:
            raise ValueError(f"Unknown backend: {backend}")

    @property
    def storage_tier(self) -> str:
        return self._backend

    def add(self, job_name: str, id_a: int, id_b: int, score: float, explanation: str) -> None:
        item = ReviewItem(job_name=job_name, id_a=id_a, id_b=id_b, score=score, explanation=explanation)
        if self._backend == "memory":
            self._items.setdefault(job_name, []).append(item)
        elif self._backend == "sqlite":
            self._sqlite_add(item)

    def list_pending(self, job_name: str) -> list[ReviewItem]:
        if self._backend == "memory":
            return [i for i in self._items.get(job_name, []) if i.status == "pending"]
        elif self._backend == "sqlite":
            return self._sqlite_list_pending(job_name)
        return []

    def approve(self, job_name: str, id_a: int, id_b: int, decided_by: str) -> None:
        for item in self._items.get(job_name, []):
            if item.id_a == id_a and item.id_b == id_b and item.status == "pending":
                item.approve(decided_by)
                return

    def reject(self, job_name: str, id_a: int, id_b: int, decided_by: str, reason: str = "") -> None:
        for item in self._items.get(job_name, []):
            if item.id_a == id_a and item.id_b == id_b and item.status == "pending":
                item.reject(decided_by, reason)
                return

    def stats(self, job_name: str) -> dict:
        items = self._items.get(job_name, []) if self._backend == "memory" else []
        return {
            "pending": sum(1 for i in items if i.status == "pending"),
            "approved": sum(1 for i in items if i.status == "approved"),
            "rejected": sum(1 for i in items if i.status == "rejected"),
        }

    # SQLite backend
    def _init_sqlite(self):
        import sqlite3
        from pathlib import Path
        db_path = Path(".goldenmatch") / "reviews.db"
        db_path.parent.mkdir(exist_ok=True)
        self._conn = sqlite3.connect(str(db_path))
        self._conn.execute("""
            CREATE TABLE IF NOT EXISTS reviews (
                job_name TEXT, id_a INTEGER, id_b INTEGER,
                score REAL, explanation TEXT, status TEXT DEFAULT 'pending',
                decided_by TEXT, decided_at TEXT, reason TEXT
            )
        """)
        self._conn.commit()

    def _sqlite_add(self, item: ReviewItem):
        self._conn.execute(
            "INSERT INTO reviews (job_name, id_a, id_b, score, explanation) VALUES (?, ?, ?, ?, ?)",
            (item.job_name, item.id_a, item.id_b, item.score, item.explanation),
        )
        self._conn.commit()

    def _sqlite_list_pending(self, job_name: str) -> list[ReviewItem]:
        rows = self._conn.execute(
            "SELECT job_name, id_a, id_b, score, explanation FROM reviews WHERE job_name = ? AND status = 'pending'",
            (job_name,),
        ).fetchall()
        return [ReviewItem(job_name=r[0], id_a=r[1], id_b=r[2], score=r[3], explanation=r[4]) for r in rows]
```

import json
from mcp.types import Tool, TextContent

from goldenmatch.core.agent import AgentSession


AGENT_TOOLS = [
    Tool(name="analyze_data", description="Profile data, detect domain, recommend strategy",
         inputSchema={"type": "object", "properties": {"file_path": {"type": "string"}}, "required": ["file_path"]}),
    Tool(name="auto_configure", description="Generate optimal matching config",
         inputSchema={"type": "object", "properties": {"file_path": {"type": "string"}, "constraints": {"type": "object"}}, "required": ["file_path"]}),
    Tool(name="agent_deduplicate", description="Run full ER pipeline with confidence gating and reasoning",
         inputSchema={"type": "object", "properties": {"file_path": {"type": "string"}, "config": {"type": "object"}}, "required": ["file_path"]}),
    # ... (8 more tools following same pattern)
]


def handle_agent_tool(name: str, arguments: dict) -> list[TextContent]:
    """Handle an agent-level MCP tool call."""
    session = AgentSession()

    if name == "analyze_data":
        result = session.analyze(arguments["file_path"])
    elif name == "agent_deduplicate":
        result = session.deduplicate(arguments["file_path"], config=arguments.get("config"))
    elif name == "auto_configure":
        result = session.auto_configure(arguments["file_path"], constraints=arguments.get("constraints"))
    # ... etc for all 10 tools
    else:
        return [TextContent(type="text", text=f"Unknown agent tool: {name}")]

    return [TextContent(type="text", text=json.dumps(result, default=str))]
```

"""CLI command for the A2A agent server."""
from __future__ import annotations

import typer

def agent_serve_cmd(
    port: int = typer.Option(8200, help="Port for the A2A server"),
    host: str = typer.Option("0.0.0.0", help="Host to bind to"),
):
    """Start the GoldenMatch A2A agent server."""
    try:
        from goldenmatch.a2a.server import create_app
        import aiohttp.web
    except ImportError:
        typer.echo("A2A server requires aiohttp. Install with: pip install goldenmatch[agent]")
        raise typer.Exit(1)

    app = create_app(host, port)
    typer.echo(f"GoldenMatch A2A agent server starting on {host}:{port}")
    typer.echo(f"Agent card: http://{host}:{port}/.well-known/agent.json")
    aiohttp.web.run_app(app, host=host, port=port)
```

def compare_strategies(self, file_path: str, ground_truth: str | None = None) -> dict:
    df = pl.read_csv(file_path, ignore_errors=True, encoding="utf8-lossy")
    profile = profile_for_agent(df)
    strategies_to_try = ["exact_only", "fuzzy"]
    # Only add F-S if there are fuzzy-matchable fields
    if any(f.uniqueness < 0.90 for f in profile.fields if f.type == "string"):
        strategies_to_try.append("fellegi_sunter")

    results = {}
    for s in strategies_to_try:
        config = self._build_config_for_strategy(s, profile)
        try:
            result = dedupe_df(df, config=config)
            _, review, _ = gate_pairs(result.scored_pairs)
            metrics = {
                "clusters": result.total_clusters,
                "match_rate": result.match_rate,
                "review_queue_size": len(review),
            }
            if ground_truth:
                from goldenmatch.core.evaluate import evaluate_clusters, load_ground_truth_csv
                gt_pairs = load_ground_truth_csv(ground_truth)
                eval_result = evaluate_clusters(result.clusters, gt_pairs)
                metrics.update(eval_result.summary())
            results[s] = metrics
        except Exception as e:
            results[s] = {"error": str(e)}
    return results
```
